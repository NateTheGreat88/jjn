// Firestore Security Rules for JJN Chat
// Paste these in the Firestore "Rules" tab in Firebase Console

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to read and write messages
    // For production, you might want to add authentication
    match /messages/{messageId} {
      allow read, write: if true;
    }
    
    // Allow anyone to read and write room messages (private rooms)
    match /roomMessages/{messageId} {
      allow read, write: if true;
    }
    
    // Allow anyone to read and write room metadata
    match /rooms/{roomId} {
      allow read, write: if true;
    }
    
    // User data - users can only read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
  }
}
